224 /**
225  * @brief Type of an USB endpoint configuration structure.
226  * @note Platform specific restrictions may apply to endpoints.
227  */
228 typedef struct {
229  /**
230  * @brief Type and mode of the endpoint.
231  */
232  uint32_t ep_mode;
233  /**
234  * @brief Setup packet notification callback.
235  * @details This callback is invoked when a setup packet has been
236  * received.
237  * @post The application must immediately call @p usbReadPacket() in
238  * order to access the received packet.
239  * @note This field is only valid for @p USB_EP_MODE_TYPE_CTRL
240  * endpoints, it should be set to @p NULL for other endpoint
241  * types.
242  */
243  usbepcallback_t setup_cb;
244  /**
245  * @brief IN endpoint notification callback.
246  * @details This field must be set to @p NULL if callback is not required.
247  */
248  usbepcallback_t in_cb;
249  /**
250  * @brief OUT endpoint notification callback.
251  * @details This field must be set to @p NULL if callback is not required.
252  */
253  usbepcallback_t out_cb;
254  /**
255  * @brief IN endpoint maximum packet size.
256  * @details This field must be set to zero if the IN endpoint is not used.
257  */
258  uint16_t in_maxsize;
259  /**
260  * @brief OUT endpoint maximum packet size.
261  * @details This field must be set to zero if the OUT endpoint is not used.
262  */
263  uint16_t out_maxsize;
264  /**
265  * @brief @p USBEndpointState associated to the IN endpoint.
266  * @details This field must be set to @p NULL if the IN endpoint is not
267  * used.
268  */
269  USBInEndpointState *in_state;
270  /**
271  * @brief @p USBEndpointState associated to the OUT endpoint.
272  * @details This field must be set to @p NULL if the OUT endpoint is not
273  * used.
274  */
275  USBOutEndpointState *out_state;
276  /* End of the mandatory fields.*/
277  /**
278  * @brief Reserved field, not currently used.
279  * @note Initialize this field to 1 in order to be forward compatible.
280  */
281  uint16_t ep_buffers;
282  /**
283  * @brief Pointer to a buffer for setup packets.
284  * @details Setup packets require a dedicated 8-bytes buffer, set this
285  * field to @p NULL for non-control endpoints.
286  */
287  uint8_t *setup_buf;
288 } USBEndpointConfig;

317 /**
318  * @brief Structure representing an USB driver.
319  */
320 struct USBDriver {
321  /**
322  * @brief Driver state.
323  */
324  usbstate_t state;
325  /**
326  * @brief Current configuration data.
327  */
328  const USBConfig *config;
329  /**
330  * @brief Bit map of the transmitting IN endpoints.
331  */
332  uint16_t transmitting;
333  /**
334  * @brief Bit map of the receiving OUT endpoints.
335  */
336  uint16_t receiving;
337  /**
338  * @brief Active endpoints configurations.
339  */
340  const USBEndpointConfig *epc[USB_MAX_ENDPOINTS + 1];
341  /**
342  * @brief Fields available to user, it can be used to associate an
343  * application-defined handler to an IN endpoint.
344  * @note The base index is one, the endpoint zero does not have a
345  * reserved element in this array.
346  */
347  void *in_params[USB_MAX_ENDPOINTS];
348  /**
349  * @brief Fields available to user, it can be used to associate an
350  * application-defined handler to an OUT endpoint.
351  * @note The base index is one, the endpoint zero does not have a
352  * reserved element in this array.
353  */
354  void *out_params[USB_MAX_ENDPOINTS];
355  /**
356  * @brief Endpoint 0 state.
357  */
358  usbep0state_t ep0state;
359  /**
360  * @brief Next position in the buffer to be transferred through endpoint 0.
361  */
362  uint8_t *ep0next;
363  /**
364  * @brief Number of bytes yet to be transferred through endpoint 0.
365  */
366  size_t ep0n;
367  /**
368  * @brief Endpoint 0 end transaction callback.
369  */
370  usbcallback_t ep0endcb;
371  /**
372  * @brief Setup packet buffer.
373  */
374  uint8_t setup[8];
375  /**
376  * @brief Current USB device status.
377  */
378  uint16_t status;
379  /**
380  * @brief Assigned USB address.
381  */
382  uint8_t address;
383  /**
384  * @brief Current USB device configuration.
385  */
386  uint8_t configuration;
387  /**
388  * @brief State of the driver when a suspend happened.
389  */
390  usbstate_t saved_state;
391 #if defined(USB_DRIVER_EXT_FIELDS)
392  USB_DRIVER_EXT_FIELDS
393 #endif
394  /* End of the mandatory fields.*/
395  /**
396  * @brief Pointer to the next address in the packet memory.
397  */
398  uint32_t pmnext;
399 };

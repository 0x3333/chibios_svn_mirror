124 /**
125  * @brief Slave address setup.
126  * @note The RW bit is set to zero internally.
127  *
128  * @param[in] i2cp pointer to the @p I2CDriver object
129  * @param[in] addr slave device address
130  *
131  * @notapi
132  */
133 static void i2c_lld_set_address(I2CDriver *i2cp, i2caddr_t addr) {
134  I2C_TypeDef *dp = i2cp->i2c;
135 
136  /* Address alignment depends on the addressing mode selected.*/
137  if ((i2cp->config->cr2 & I2C_CR2_ADD10) == 0U)
138  dp->CR2 = (uint32_t)addr << 1U;
139  else
140  dp->CR2 = (uint32_t)addr;
141 }

143 /**
144  * @brief I2C RX transfer setup.
145  *
146  * @param[in] i2cp pointer to the @p I2CDriver object
147  *
148  * @notapi
149  */
150 static void i2c_lld_setup_rx_transfer(I2CDriver *i2cp) {
151  I2C_TypeDef *dp = i2cp->i2c;
152  uint32_t reload;
153  size_t n;
154 
155  /* The unit can transfer 255 bytes maximum in a single operation.*/
156  n = i2c_lld_get_rxbytes(i2cp);
157  if (n > 255U) {
158  n = 255U;
159  reload = I2C_CR2_RELOAD;
160  }
161  else {
162  reload = 0U;
163  }
164 
165  /* Configures the CR2 registers with both the calculated and static
166  settings.*/
167  dp->CR2 = (dp->CR2 & ~(I2C_CR2_NBYTES | I2C_CR2_RELOAD)) | i2cp->config->cr2 |
168  I2C_CR2_RD_WRN | (n << 16U) | reload;
169 }
170 
171 /**
172  * @brief I2C TX transfer setup.
173  *
174  * @param[in] i2cp pointer to the @p I2CDriver object
175  *
176  * @notapi
177  */
178 static void i2c_lld_setup_tx_transfer(I2CDriver *i2cp) {
179  I2C_TypeDef *dp = i2cp->i2c;
180  uint32_t reload;
181  size_t n;
182 
183  /* The unit can transfer 255 bytes maximum in a single operation.*/
184  n = i2c_lld_get_txbytes(i2cp);
185  if (n > 255U) {
186  n = 255U;
187  reload = I2C_CR2_RELOAD;
188  }
189  else {
190  reload = 0U;
191  }
192 
193  /* Configures the CR2 registers with both the calculated and static
194  settings.*/
195  dp->CR2 = (dp->CR2 & ~(I2C_CR2_NBYTES | I2C_CR2_RELOAD)) | i2cp->config->cr2 |
196  (n << 16U) | reload;
197 }

199 /**
200  * @brief Aborts an I2C transaction.
201  *
202  * @param[in] i2cp pointer to the @p I2CDriver object
203  *
204  * @notapi
205  */
206 static void i2c_lld_abort_operation(I2CDriver *i2cp) {
207  I2C_TypeDef *dp = i2cp->i2c;
208 
209  if (dp->CR1 & I2C_CR1_PE) {
210  /* Stops the I2C peripheral.*/
211  dp->CR1 &= ~I2C_CR1_PE;
212  while (dp->CR1 & I2C_CR1_PE)
213  dp->CR1 &= ~I2C_CR1_PE;
214  dp->CR1 |= I2C_CR1_PE;
215  }
216 
217 #if STM32_I2C_USE_DMA == TRUE
218  /* Stops the associated DMA streams.*/
219  dmaStreamDisable(i2cp->dmatx);
220  dmaStreamDisable(i2cp->dmarx);
221 #else
222  dp->CR1 &= ~(I2C_CR1_TXIE | I2C_CR1_RXIE);
223 #endif
224 }

226 /**
227  * @brief I2C shared ISR code.
228  *
229  * @param[in] i2cp pointer to the @p I2CDriver object
230  * @param[in] isr content of the ISR register to be decoded
231  *
232  * @notapi
233  */
234 static void i2c_lld_serve_interrupt(I2CDriver *i2cp, uint32_t isr) {
235  I2C_TypeDef *dp = i2cp->i2c;
236 
237  /* Special case of a received NACK, the transfer is aborted.*/
238  if ((isr & I2C_ISR_NACKF) != 0U) {
239 #if STM32_I2C_USE_DMA == TRUE
240  /* Stops the associated DMA streams.*/
241  dmaStreamDisable(i2cp->dmatx);
242  dmaStreamDisable(i2cp->dmarx);
243 #endif
244 
245  /* Error flag.*/
246  i2cp->errors |= I2C_ACK_FAILURE;
247 
248  /* Transaction finished sending the STOP.*/
249  dp->CR2 |= I2C_CR2_STOP;
250 
251  /* Make sure no more interrupts.*/
252  dp->CR1 &= ~(I2C_CR1_TCIE | I2C_CR1_TXIE | I2C_CR1_RXIE);
253 
254  /* Errors are signaled to the upper layer.*/
255  _i2c_wakeup_error_isr(i2cp);
256 
257  return;
258  }
259 
260 #if STM32_I2C_USE_DMA == FALSE
261  /* Handling of data transfer if the DMA mode is disabled.*/
262  {
263  uint32_t cr1 = dp->CR1;
264 
265  if (i2cp->state == I2C_ACTIVE_TX) {
266  /* Transmission phase.*/
267  if (((cr1 &I2C_CR1_TXIE) != 0U) && ((isr & I2C_ISR_TXIS) != 0U)) {
268  dp->TXDR = (uint32_t)*i2cp->txptr;
269  i2cp->txptr++;
270  i2cp->txbytes--;
271  if (i2cp->txbytes == 0U) {
272  dp->CR1 &= ~I2C_CR1_TXIE;
273  }
274  }
275  }
276  else {
277  /* Receive phase.*/
278  if (((cr1 & I2C_CR1_RXIE) != 0U) && ((isr & I2C_ISR_RXNE) != 0U)) {
279  *i2cp->rxptr = (uint8_t)dp->RXDR;
280  i2cp->rxptr++;
281  i2cp->rxbytes--;
282  if (i2cp->rxbytes == 0U) {
283  dp->CR1 &= ~I2C_CR1_RXIE;
284  }
285  }
286  }
287  }
288 #endif
289 
290  /* Partial transfer handling, restarting the transfer and returning.*/
291  if ((isr & I2C_ISR_TCR) != 0U) {
292  if (i2cp->state == I2C_ACTIVE_TX) {
293  i2c_lld_setup_tx_transfer(i2cp);
294  }
295  else {
296  i2c_lld_setup_rx_transfer(i2cp);
297  }
298  return;
299  }
300 
301  /* The following condition is true if a transfer phase has been completed.*/
302  if ((isr & I2C_ISR_TC) != 0U) {
303  if (i2cp->state == I2C_ACTIVE_TX) {
304  /* End of the transmit phase.*/
305 
306 #if STM32_I2C_USE_DMA == TRUE
307  /* Disabling TX DMA channel.*/
308  dmaStreamDisable(i2cp->dmatx);
309 #endif
310 
311  /* Starting receive phase if necessary.*/
312  if (i2c_lld_get_rxbytes(i2cp) > 0U) {
313  /* Setting up the peripheral.*/
314  i2c_lld_setup_rx_transfer(i2cp);
315 
316 #if STM32_I2C_USE_DMA == TRUE
317  /* Enabling RX DMA.*/
318  dmaStreamEnable(i2cp->dmarx);
319 #else
320  /* RX interrupt enabled.*/
321  dp->CR1 |= I2C_CR1_RXIE;
322 #endif
323 
324  /* Starts the read operation.*/
325  dp->CR2 |= I2C_CR2_START;
326 
327  /* State change.*/
328  i2cp->state = I2C_ACTIVE_RX;
329 
330  /* Note, returning because the transaction is not over yet.*/
331  return;
332  }
333  }
334  else {
335  /* End of the receive phase.*/
336 #if STM32_I2C_USE_DMA == TRUE
337  /* Disabling RX DMA channel.*/
338  dmaStreamDisable(i2cp->dmarx);
339 #endif
340  }
341 
342  /* Transaction finished sending the STOP.*/
343  dp->CR2 |= I2C_CR2_STOP;
344 
345  /* Make sure no more 'Transfer Complete' interrupts.*/
346  dp->CR1 &= ~I2C_CR1_TCIE;
347 
348  /* Normal transaction end.*/
349  _i2c_wakeup_isr(i2cp);
350  }
351 }

353 /**
354  * @brief I2C error handler.
355  *
356  * @param[in] i2cp pointer to the @p I2CDriver object
357  * @param[in] isr content of the ISR register to be decoded
358  *
359  * @notapi
360  */
361 static void i2c_lld_serve_error_interrupt(I2CDriver *i2cp, uint32_t isr) {
362 
363 #if STM32_I2C_USE_DMA == TRUE
364  /* Clears DMA interrupt flags just to be safe.*/
365  dmaStreamDisable(i2cp->dmatx);
366  dmaStreamDisable(i2cp->dmarx);
367 #else
368  /* Disabling RX and TX interrupts.*/
369  i2cp->i2c->CR1 &= ~(I2C_CR1_TXIE | I2C_CR1_RXIE);
370 #endif
371 
372  if (isr & I2C_ISR_BERR)
373  i2cp->errors |= I2C_BUS_ERROR;
374 
375  if (isr & I2C_ISR_ARLO)
376  i2cp->errors |= I2C_ARBITRATION_LOST;
377 
378  if (isr & I2C_ISR_OVR)
379  i2cp->errors |= I2C_OVERRUN;
380 
381  if (isr & I2C_ISR_TIMEOUT)
382  i2cp->errors |= I2C_TIMEOUT;
383 
384  /* If some error has been identified then sends wakes the waiting thread.*/
385  if (i2cp->errors != I2C_NO_ERROR)
386  _i2c_wakeup_error_isr(i2cp);
387 }

617 /**
618  * @brief Low level I2C driver initialization.
619  *
620  * @notapi
621  */
622 void i2c_lld_init(void) {
623 
624 #if STM32_I2C_USE_I2C1

942  * @brief Receives data via the I2C bus as master.
943  *
944  * @param[in] i2cp pointer to the @p I2CDriver object
945  * @param[in] addr slave device address
946  * @param[out] rxbuf pointer to the receive buffer
947  * @param[in] rxbytes number of bytes to be received
948  * @param[in] timeout the number of ticks before the operation timeouts,
949  * the following special values are allowed:
950  * - @a TIME_INFINITE no timeout.
951  * .
952  * @return The operation status.
953  * @retval MSG_OK if the function succeeded.
954  * @retval MSG_RESET if one or more I2C errors occurred, the errors can
955  * be retrieved using @p i2cGetErrors().
956  * @retval MSG_TIMEOUT if a timeout occurred before operation end. <b>After a
957  * timeout the driver must be stopped and restarted
958  * because the bus is in an uncertain state</b>.
959  *
960  * @notapi
961  */
962 msg_t i2c_lld_master_receive_timeout(I2CDriver *i2cp, i2caddr_t addr,
963  uint8_t *rxbuf, size_t rxbytes,
964  sysinterval_t timeout) {
965  msg_t msg;
966  I2C_TypeDef *dp = i2cp->i2c;
967  systime_t start, end;
968 
969  /* Resetting error flags for this transfer.*/
970  i2cp->errors = I2C_NO_ERROR;
971 
972  /* Releases the lock from high level driver.*/
973  osalSysUnlock();
974 
975 #if STM32_I2C_USE_DMA == TRUE
976  /* RX DMA setup.*/
977  dmaStreamSetMode(i2cp->dmarx, i2cp->rxdmamode);
978  dmaStreamSetMemory0(i2cp->dmarx, rxbuf);
979  dmaStreamSetTransactionSize(i2cp->dmarx, rxbytes);
980 #else
981  i2cp->rxptr = rxbuf;
982  i2cp->rxbytes = rxbytes;
983 #endif
984 
985  /* Calculating the time window for the timeout on the busy bus condition.*/
986  start = osalOsGetSystemTimeX();
987  end = osalTimeAddX(start, OSAL_MS2I(STM32_I2C_BUSY_TIMEOUT));
988 
989  /* Waits until BUSY flag is reset or, alternatively, for a timeout
990  condition.*/
991  while (true) {
992  osalSysLock();
993 
994  /* If the bus is not busy then the operation can continue, note, the
995  loop is exited in the locked state.*/
996  if ((dp->ISR & I2C_ISR_BUSY) == 0)
997  break;
998 
999  /* If the system time went outside the allowed window then a timeout
1000  condition is returned.*/
1001  if (!osalTimeIsInRangeX(osalOsGetSystemTimeX(), start, end)) {
1002  return MSG_TIMEOUT;
1003  }
1004 
1005  osalSysUnlock();
1006  }
1007 
1008  /* Setting up the slave address.*/
1009  i2c_lld_set_address(i2cp, addr);
1010 
1011  /* Setting up the peripheral.*/
1012  i2c_lld_setup_rx_transfer(i2cp);
1013 
1014 #if STM32_I2C_USE_DMA == TRUE
1015  /* Enabling RX DMA.*/
1016  dmaStreamEnable(i2cp->dmarx);
1017 
1018  /* Transfer complete interrupt enabled.*/
1019  dp->CR1 |= I2C_CR1_TCIE;
1020 #else
1021 
1022  /* Transfer complete and RX interrupts enabled.*/
1023  dp->CR1 |= I2C_CR1_TCIE | I2C_CR1_RXIE;
1024 #endif
1025 
1026  /* Starts the operation.*/
1027  dp->CR2 |= I2C_CR2_START;
1028 
1029  /* Waits for the operation completion or a timeout.*/
1030  msg = osalThreadSuspendTimeoutS(&i2cp->thread, timeout);
1031 
1032  /* In case of a software timeout a STOP is sent as an extreme attempt
1033  to release the bus.*/
1034  if (msg == MSG_TIMEOUT) {
1035  dp->CR2 |= I2C_CR2_STOP;
1036  }
1037 
1038  return msg;
1039 }
